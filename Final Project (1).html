#Question 4 How does the impact of smoking on blood pressure vary across different age groups? 

if 'age_group' not in data.columns:
    data['age_group'] = pd.cut(data['age'], bins=[30, 40, 50, 60, 70], labels=['30-40', '40-50', '50-60', '60-70'])

# Group data by age group and smoking status, then calculate average blood pressure
bp_by_age_smoking = data.groupby(['age_group', 'current_smoker'])[['systolic_bp', 'diastolic_bp']].mean()

# Visualizing the results
import matplotlib.pyplot as plt
import seaborn as sns

plt.figure(figsize=(12, 6))
sns.lineplot(data=bp_by_age_smoking, x='age_group', y='systolic_bp', hue='current_smoker', marker='o')
plt.title('Average Systolic Blood Pressure by Age Group and Smoking Status')
plt.xlabel('Age Group')
plt.ylabel('Average Systolic Blood Pressure')
plt.legend(title='Smoker Status')
plt.grid(True)
plt.show()

plt.figure(figsize=(12, 6))
